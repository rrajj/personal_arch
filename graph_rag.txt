from google.adk.agents import Agent
from google.adk.tools import FunctionTool
import psycopg2
from psycopg2.extras import RealDictCursor

# Tool: execute any dynamic SQL query
def execute_query(sql: str, params: list = []):
    """
    Executes a dynamic SQL query against the permit graph database.
    Returns results as a list of dictionaries.
    """
    try:
        conn = get_db_connection()
        cur = conn.cursor(cursor_factory=RealDictCursor)
        cur.execute(sql, params)
        results = cur.fetchall()
        conn.close()
        return {"success": True, "data": results, "row_count": len(results)}
    except Exception as e:
        return {"success": False, "error": str(e)}


# Tool: get schema context so agent knows what tables/columns exist
def get_schema_context():
    """Returns the database schema for the agent to understand available tables."""
    return """
    TABLE: arb_graph_nodes
    - id          UUID        → unique node identifier
    - type        TEXT        → 'permit' or 'pts'
    - timestamp   TIMESTAMPTZ → when node was created
    - is_deleted  BOOLEAN     → soft delete flag
    - properties  JSONB       → flexible data storage:
                                 For permits: { permit_number, permit_type, 
                                               status, approved_at, 
                                               submitted_at, expires_at }
                                 For pts:     { pts_id, name }
    - embedding_text TEXT     → raw text used to generate embedding
    - embedding   VECTOR      → semantic vector
    - tsv         TSVECTOR    → for full text search

    TABLE: arb_graph_edges
    - id          UUID  → unique edge identifier
    - source_id   UUID  → references arb_graph_nodes.id (usually pts node)
    - target_id   UUID  → references arb_graph_nodes.id (usually permit node)
    - relation    TEXT  → e.g. 'HAS_PERMIT'
    - properties  JSONB → extra edge metadata
    - page_id     UUID  → optional page reference

    TABLE: arb_permit_embeddings
    - id          UUID        → unique embedding identifier
    - permit_id   UUID        → references arb_graph_nodes.id (permit node)
    - pts_id      UUID        → references arb_graph_nodes.id (pts node)
    - permit_type TEXT        → 'to_design', 'to_build', 'to_deploy'
    - text        TEXT        → full text description of permit
    - embedding   VECTOR      → semantic vector for similarity search
    - timestamp   TIMESTAMPTZ → when embedding was created
    - is_deleted  BOOLEAN     → soft delete flag
    - tsv         TSVECTOR    → for full text search
    """

# SQL Agent
sql_agent = Agent(
    name="sql_agent",
    model="gemini-2.0-flash",
    description="Writes and executes dynamic SQL queries against the permit graph database.",
    instruction="""
        You are a PostgreSQL expert that writes dynamic SQL queries.
        
        You have access to these tools:
        - get_schema_context: call this first to understand the tables
        - execute_query: runs the SQL you write
        
        Rules:
        1. ALWAYS call get_schema_context first before writing any query
        2. Write safe, read-only SELECT queries only — never INSERT, UPDATE, DELETE
        3. Use properties->>'field_name' to access JSONB fields
        4. For semantic search use: embedding <=> '[vector]'::vector
        5. For keyword search use: tsv @@ to_tsquery('keyword')
        6. Always filter is_deleted = false
        7. Return clear, structured results
        
        Example queries you can write:
        
        -- Find all permits for a PTS
        SELECT 
            source.properties->>'pts_id'        AS pts_id,
            target.properties->>'permit_number'  AS permit_number,
            target.properties->>'permit_type'    AS permit_type,
            target.properties->>'status'         AS status,
            target.properties->>'approved_at'    AS approved_at
        FROM arb_graph_edges e
        JOIN arb_graph_nodes source ON source.id = e.source_id
        JOIN arb_graph_nodes target ON target.id = e.target_id
        WHERE source.properties->>'pts_id' = 'PTS-001'
          AND source.is_deleted = false
          AND target.is_deleted = false;
    """,
    tools=[
        FunctionTool(get_schema_context),
        FunctionTool(execute_query)
    ]
)

import openai

# Tool: embed the user query
def embed_query(text: str):
    """Converts user query text into a vector embedding."""
    response = openai.embeddings.create(
        input=text,
        model="text-embedding-3-small"
    )
    return response.data[0].embedding


# Tool: vector search on arb_permit_embeddings
def vector_search(query_embedding: list, permit_type: str = None, limit: int = 5):
    """
    Finds most semantically similar permits using embedding similarity.
    This is the RAG retrieval step.
    """
    conn = get_db_connection()
    cur = conn.cursor(cursor_factory=RealDictCursor)

    sql = """
        SELECT 
            permit_id,
            pts_id,
            permit_type,
            text,
            embedding <=> %s::vector AS distance
        FROM arb_permit_embeddings
        WHERE is_deleted = false
        {filter}
        ORDER BY distance
        LIMIT %s
    """.format(filter="AND permit_type = %s" if permit_type else "")

    params = [query_embedding, permit_type, limit] if permit_type else [query_embedding, limit]
    cur.execute(sql, params)
    return cur.fetchall()


# Tool: graph expansion from retrieved permit ids
def expand_graph(permit_ids: list):
    """
    Given permit node ids found via vector search,
    expand the graph to get connected PTS nodes and edge metadata.
    This is the GRAPH part of Graph RAG.
    """
    conn = get_db_connection()
    cur = conn.cursor(cursor_factory=RealDictCursor)

    sql = """
        SELECT 
            source.properties   AS pts_properties,
            target.properties   AS permit_properties,
            e.relation,
            e.properties        AS edge_properties
        FROM arb_graph_edges e
        JOIN arb_graph_nodes source ON source.id = e.source_id
        JOIN arb_graph_nodes target ON target.id = e.target_id
        WHERE target.id = ANY(%s::uuid[])
          AND source.is_deleted = false
          AND target.is_deleted = false
    """
    cur.execute(sql, [permit_ids])
    return cur.fetchall()


# Embedding Agent
embedding_agent = Agent(
    name="embedding_agent",
    model="gemini-2.0-flash",
    description="Handles semantic retrieval using embeddings then expands via graph.",
    instruction="""
        You are a Graph RAG retrieval agent. Follow these steps strictly:

        STEP 1 — Embed the user query
            Call embed_query(text) with the user's question

        STEP 2 — Vector search (RAG retrieval)
            Call vector_search(query_embedding) with the embedding from step 1
            This finds the most semantically relevant permits

        STEP 3 — Graph expansion
            Take the permit_ids from step 2
            Call expand_graph(permit_ids) to get connected PTS nodes
            This enriches the context with graph relationships

        STEP 4 — Return enriched context
            Combine vector search results + graph expansion results
            Return as structured context for the root agent to answer from

        This is Graph RAG:
        - Vector search = RAG (finds relevant nodes semantically)
        - Graph expansion = Graph (enriches with connected context)
    """,
    tools=[
        FunctionTool(embed_query),
        FunctionTool(vector_search),
        FunctionTool(expand_graph)
    ]
)
